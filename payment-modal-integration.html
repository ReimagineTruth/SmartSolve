Custom Payment Handling

// Custom success handler paymentModal.onSuccess((details) => { // Send to analytics gtag('event', 'purchase', { value: details.amount, currency: 'Pi', plan: details.plan }); // Update user status updateUserSubscription(details); // Show success message showSuccessMessage('Welcome to SmartSolve!'); }); // Custom cancel handler paymentModal.onCancel(() => { // Track abandonment gtag('event', 'payment_cancelled'); // Show alternative options showAlternativePlans(); });

Dynamic Plan Selection

// Open modal with specific plan based on user action function handleUpgradeClick(planType) { // Track the upgrade attempt analytics.track('upgrade_clicked', { plan: planType }); // Open payment modal openPaymentModal(planType); } // Usage <button onclick="handleUpgradeClick('premium')"> Upgrade to Premium </button>

Integration with Pi SDK

// Custom payment processing with Pi SDK paymentModal.onSuccess((details) => { // Initialize Pi payment Pi.createPayment({ amount: details.amount.replace(' Pi', ''), memo: `SmartSolve ${details.plan} Plan`, metadata: { plan: details.plan, billing: details.billing } }).then((payment) => { // Handle successful payment console.log('Pi payment successful:', payment); redirectToDashboard(); }).catch((error) => { console.error('Payment failed:', error); showErrorMessage('Payment failed. Please try again.'); }); });
